    nom: str
    prenom: str
    email: str
    role: str # etudiant, enseignant, admin

@app.get("/")
def admin_root():
    return {"message": "Service d'Administration (Gestion Utilisateurs)"}

@app.post("/users/")
def creer_utilisateur(user: User, token_data: dict = Depends(verifier_token)):
    # 1. Validation RBAC stricte : Rôle Admin Requis
    roles = token_data.get("realm_access", {}).get("roles", [])
    if "admin" not in roles:
        raise HTTPException(status_code=403, detail="Accès refusé : rôle administrateur requis")
    # 2. Création de l'utilisateur dans Keycloak via l'API Admin REST
    try:
        new_user_id = keycloak_admin.create_user({
            "email": user.email,
            "username": user.email,
            "enabled": True,
            "firstName": user.prenom,
            "lastName": user.nom,
            "credentials": [{"value": "admin", "type": "password", "temporary": False}] # Mot de passe par défaut
        }, exist_ok=False)
    except KeycloakError as e:
        raise HTTPException(status_code=400, detail=f"Erreur lors de la création coté Keycloak : {e}")
    
    # 2.5 Assigner le rôle métier au nouvel utilisateur
    try:
        try:
            role = keycloak_admin.get_realm_role(user.role)
        except KeycloakError:
            keycloak_admin.create_realm_role({"name": user.role})
            role = keycloak_admin.get_realm_role(user.role)
        keycloak_admin.assign_realm_roles(user_id=new_user_id, roles=[role])
    except Exception as e:
        print(f"Erreur d'assignement de rôle : {e}")
        # On ne bloque pas si le rôle échoue, mais on le log
        
    # 3. Sauvegarde d'informations complémentaires dans Cassandra
    try:
        session = get_db_session()
        query = "INSERT INTO utilisateurs (email, nom, prenom, role) VALUES (%s, %s, %s, %s)"
        session.execute(query, (user.email, user.nom, user.prenom, user.role))
    except Exception as dbe:
        raise HTTPException(status_code=500, detail=f"Erreur Cassandra : {dbe}")
    
    return {"message": "Utilisateur créé avec succès dans Keycloak et Cassandra", "user": user}

@app.get("/cours/download/{fichier_id}")
def telecharger_cours(fichier_id: str, token_data: dict = Depends(verifier_token)):

    if "etudiant" not in roles:
        raise HTTPException(status_code=403, detail="Accès refusé : rôle étudiant requis")

    try:
        f_id = uuid.UUID(fichier_id)
        resultat = session.execute("SELECT nom_fichier FROM cours WHERE id = %s", [f_id])
        if resultat.one():
            fichier_nom = resultat.one().nom_fichier
            
            # Génération d'un lien sécurisé (valable 2 heures max par exemple)
            # timedelta(hours=2) is used by minio python library
            from datetime import timedelta
            url_telechargement = minio_client.presigned_get_object("cours", fichier_nom, expires=timedelta(hours=2))
            
            return {"url_telechargement": url_telechargement}
        else:
            raise HTTPException(status_code=404, detail="Fichier non trouvé")
    except ValueError:
        raise HTTPException(status_code=400, detail="ID invalide")

from cassandra.cluster import Cluster
from cassandra.query import SimpleStatement

def verifier_connexion():
    print("Tentative de connexion à Cassandra...")
    cluster = Cluster(['127.0.0.1'], port=9042)
    
    # On va essayer de se connecter avec un retry pour attendre que Cassandra démarre
    session = None
    for i in range(20):
        try:
            session = cluster.connect()
            print("Connecté avec succès à Cassandra !")
            break
        except Exception as e:
            print(f"En attente de Cassandra... ({i+1}/20)")
            time.sleep(2)
            
    if not session:
        print("Impossible de se connecter à Cassandra. Vérifiez que Docker est bien lancé.")
        exit(1)
        
    return cluster, session

def initialiser_base(session):
    print("Création du Keyspace 'ent_db'...")
    session.execute("""
        CREATE KEYSPACE IF NOT EXISTS ent_db 
        WITH replication = {'class':'SimpleStrategy', 'replication_factor':1};
    """)
    
    session.set_keyspace('ent_db')

    print("Création de la table 'utilisateurs'...")
    session.execute("""
        CREATE TABLE IF NOT EXISTS utilisateurs (
            email text PRIMARY KEY,
            nom text,
            prenom text,
            role text
        );
    """)

    print("Création de la table 'cours'...")
    session.execute("""
        CREATE TABLE IF NOT EXISTS cours (
            id uuid PRIMARY KEY,
            titre text,
            description text,
            nom_fichier text,
            url text
        );
    """)

    print("Initialisation terminée avec succès !")

if __name__ == "__main__":
    cluster, session = verifier_connexion()
    initialiser_base(session)
    cluster.shutdown()
